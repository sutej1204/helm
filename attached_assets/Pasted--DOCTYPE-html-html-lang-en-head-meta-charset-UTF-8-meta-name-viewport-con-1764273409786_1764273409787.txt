    <script>
        // --- Modal Functions ---
        const emailModal = document.getElementById('email-modal');
        const modalContent = document.getElementById('modal-content');
        const copyMessage = document.getElementById('copy-message');

        function draftEmail(exceptionType, amount) {
            const emailBody = `
                Dear Vendor Team,

                Vista's Financial OS has identified a variance in the recent credit memo processing related to your account (GTS-1004A).

                **Exception Type:** ${exceptionType}
                **Discrepancy Amount:** -$${amount.toFixed(2)}

                Vista's analysis indicates we were under-credited by this amount based on our mutual program agreements and the following verified sales data (Attached CSV/PDF):

                1. **Program Rate:** 8.5%
                2. **Reconciled Sales Volume:** $1,532,761.00
                3. **Expected Rebate:** $122,620.88

                Please review the attached reconciliation document and process the necessary correction to resolve this $${amount.toFixed(2)} discrepancy.

                Best regards,
                Vista Financial Operations
            `;
            
            modalContent.innerHTML = emailBody.replace(/\n/g, '<br>').replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>');
            emailModal.classList.remove('hidden');
        }

        function closeModal() {
            emailModal.classList.add('hidden');
            copyMessage.classList.add('opacity-0');
        }

        function copyToClipboard() {
            // Get the plain text from the modal content
            const textToCopy = modalContent.innerText;
            
            // Use execCommand for broader iframe compatibility
            const tempInput = document.createElement('textarea');
            tempInput.value = textToCopy;
            document.body.appendChild(tempInput);
            tempInput.select();
            
            try {
                document.execCommand('copy');
                copyMessage.classList.remove('opacity-0');
                setTimeout(() => copyMessage.classList.add('opacity-0'), 2000);
            } catch (err) {
                console.error('Could not copy text: ', err);
            } finally {
                document.body.removeChild(tempInput);
            }
        }
        
        // --- Task Resolution Function ---
        function resolveTask(button) {
            const taskItem = button.closest('.border-l-4');
            const taskHeader = taskItem.querySelector('.font-semibold');
            
            // Visual feedback: mark as resolved/pending
            taskItem.classList.remove('border-red-500', 'bg-red-50', 'border-orange-500', 'bg-orange-50', 'border-yellow-500', 'bg-yellow-50');
            taskItem.classList.add('border-green-500', 'bg-green-50');
            
            // Update text
            taskHeader.textContent = taskHeader.textContent.replace('(Q3 2024)', '(Resolved)');
            
            // Disable buttons
            const buttons = taskItem.querySelectorAll('button');
            buttons.forEach(btn => {
                btn.disabled = true;
                btn.classList.add('opacity-50', 'cursor-not-allowed');
                btn.classList.remove('hover:bg-indigo-700', 'hover:bg-gray-300');
            });

            // Update leakage count (simplified)
            const leakageCountElement = document.querySelector('.metric-value.text-xl.sm\\:text-2xl.font-extrabold.mt-1.text-red-700');
            let currentCount = parseInt(leakageCountElement.textContent);
            if (currentCount > 0) {
                leakageCountElement.textContent = currentCount - 1;
            }
        }

    </script>
